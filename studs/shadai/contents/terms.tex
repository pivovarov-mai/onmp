\newglossaryentry{id1}{
    name={API},
    description={это набор методов, функций и инструментов, предоставляемых разработчикам программного обеспечения для взаимодействия с определенным программным продуктом или сервисом. API определяет доступные операции, форматы данных и протоколы коммуникации, позволяя разным приложениям взаимодействовать между собой.} 
}

\newglossaryentry{id2}{
    name={Интерфейс},
    description={это точка взаимодействия между различными компонентами или системами, определяющая способы передачи данных, коммуникации и обмена информацией. Он определяет доступные операции, параметры и форматы данных, необходимые для взаимодействия между различными компонентами или системами.} 
}

\newglossaryentry{id3}{
    name={Фреймворк},
    description={это набор программных компонентов, инструментов и библиотек, предоставляемых разработчикам для ускорения и упрощения процесса создания программного обеспечения. Фреймворк устанавливает структуру и архитектурные принципы разработки, предлагает готовые решения для типовых задач и облегчает интеграцию с другими компонентами.} 
}
